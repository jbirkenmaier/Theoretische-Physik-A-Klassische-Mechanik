\documentclass{article}
\usepackage{tikz}
\usepackage{amsmath}

\begin{document}
	
	\begin{tikzpicture}
		% Draw axes
		\draw[->] (0,0) -- (3,0) node[right] {$y$}; 
		\draw[->] (0,0) -- (0,3) node[above] {$z$}; 
		\draw[->] (0,0) -- (-2,-2) node[below] {$x$}; 
		
		% Draw cylindrical point
		\coordinate (O) at (0,0); % Origin
		\coordinate (P) at (1.5,1); % Point in cylindrical coordinates
		\coordinate (ProjP) at (1.5,0); % Projection of P onto the x-y plane
		
		\draw[->, thick, red] (O) -- (P) node[above right] {$\vec{r}$}; % Radial vector (in the x-y plane)
		\draw[dashed, gray] (P) -- (ProjP) node[below right] {}; % Projection line in x-y plane
		\draw[->, dashed, gray] (ProjP) -- (0,0) node[below left] {}; % Radius in x-y plane
		
		% Add theta (polar angle)
		\draw[->, thick, blue] (0.5,0.33) arc[start angle=10, end angle=70, radius=0.7cm];
		\node[blue] at (0.2,0.5) {$\theta$};
		
		% Add z (height)
		\draw[->, thick, orange] (1.5,1) -- (1.5,0) node[midway, right] {$z$}; % Vertical projection on z-axis
		
		% Draw labels
		\filldraw[red] (P) circle (2pt); % Cylindrical point
	\end{tikzpicture}
	
\end{document}
